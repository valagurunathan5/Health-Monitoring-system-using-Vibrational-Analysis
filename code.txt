/* USER CODE BEGIN 4 */
void fill_mpu_buffer() {
  for (int i = 0; i < BUFFER_SIZE; i++) {
    MPU6050_Read_Gyro(gyro_data);  // Fill gyro_data with values
    mpu_buffer[AXIS * i] = (float)gyro_data[0];  // X-axis
    mpu_buffer[AXIS * i + 1] = (float)gyro_data[1];  // Y-axis
    mpu_buffer[AXIS * i + 2] = (float)gyro_data[2];  // Z-axis
  }
}

void Log() {
  fill_mpu_buffer();
  for (int i = 0; i < BUFFER_SIZE; i++) {
    printf("%.2f ", mpu_buffer[AXIS * i]);
    printf("%.2f ", mpu_buffer[AXIS * i + 1]);
    printf("%.2f ", mpu_buffer[AXIS * i + 2]);
  }
  printf("\r\n");
  HAL_Delay(100);
}

int __io_putchar(int ch) {
  HAL_UART_Transmit(&huart2, (uint8_t *)&ch, 1, HAL_MAX_DELAY);
  return ch;
}
